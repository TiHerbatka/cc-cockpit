# cc-cockpit MVP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a personal local web app that shows all live Claude Code sessions in one window, lets you switch between them instantly, and type into any of them.

**Architecture:** A Node server owns each `claude` session as a pseudo-terminal (`node-pty`/ConPTY) and bridges them to a single `xterm.js` browser page over a WebSocket. Core logic (ring buffer, session registry) is plain, event-emitting, dependency-injected code so it is unit-testable without spawning real processes; the PTY adapter and server wiring get integration tests with a benign command / fake PTY; the browser UI is verified manually against the spec's acceptance criteria.

**Tech Stack:** Node.js v24, `node-pty`, `ws`, `@xterm/xterm` + `@xterm/addon-fit` (served as static vendor assets), Node built-in test runner (`node --test`). No bundler/build step.

**Spec:** `docs/superpowers/specs/2026-06-16-cc-cockpit-design.md`

## Prerequisites & risks (read first)

- `node-pty` is a native module. On Windows it normally installs from prebuilt binaries, but if no prebuilt matches Node v24 it compiles from source and needs **Visual Studio Build Tools (C++ "Desktop development with C++" workload)** and **Python 3**. Task 1 includes a verification step; if `require('node-pty')` fails, install VS Build Tools + Python and re-run `npm rebuild node-pty`.
- The app is equivalent to shell access. The server binds **only** to `127.0.0.1`. Never expose the port to the network.

```js
// server/buffer.js
class RingBuffer {
  constructor(maxBytes = 64 * 1024) {
    this.maxBytes = maxBytes;
    this.data = '';
  }
  push(chunk) {
    this.data += chunk;
    if (this.data.length > this.maxBytes) {
      this.data = this.data.slice(this.data.length - this.maxBytes);
    }
  }
  getAll() {
    return this.data;
  }
}
module.exports = { RingBuffer };
```

```js
// server/sessions.js
const { EventEmitter } = require('node:events');
const crypto = require('node:crypto');
const path = require('node:path');
const { RingBuffer } = require('./buffer');

const IDLE_AFTER_MS = 2000;

class SessionRegistry extends EventEmitter {
  constructor({ spawnPty, now = () => Date.now() }) {
    super();
    this.spawnPty = spawnPty;
    this.now = now;
    this.sessions = new Map();
  }

  create(cwd) {
    const id = crypto.randomUUID();
    const label = path.basename(cwd) || cwd;
    const pty = this.spawnPty(cwd);
    const session = { id, cwd, label, status: 'working', lastOut: this.now(), buffer: new RingBuffer(), pty };
    this.sessions.set(id, session);
    pty.onData((data) => this.appendOutput(id, data));
    pty.onExit(() => this.markExited(id));
    this.emit('sessions');
    return this._public(session);
  }

  appendOutput(id, data) {
    const s = this.sessions.get(id);
    if (!s || s.status === 'exited') return;
    s.buffer.push(data);
    s.lastOut = this.now();
    const wasNotWorking = s.status !== 'working';
    s.status = 'working';
    this.emit('output', id, data);
    if (wasNotWorking) this.emit('sessions');
  }

  write(id, data) {
    const s = this.sessions.get(id);
    if (s && s.status !== 'exited') s.pty.write(data);
  }

  markExited(id) {
    const s = this.sessions.get(id);
    if (s && s.status !== 'exited') {
      s.status = 'exited';
      this.emit('sessions');
    }
  }

  tickStatus() {
    const t = this.now();
    let changed = false;
    for (const s of this.sessions.values()) {
      if (s.status === 'working' && t - s.lastOut >= IDLE_AFTER_MS) {
        s.status = 'idle';
        changed = true;
      }
    }
    if (changed) this.emit('sessions');
  }

  get(id) {
    const s = this.sessions.get(id);
    return s ? this._public(s) : null;
  }

  bufferOf(id) {
    const s = this.sessions.get(id);
    return s ? s.buffer.getAll() : '';
  }

  list() {
    return [...this.sessions.values()].map((s) => this._public(s));
  }

  _public(s) {
    return { id: s.id, cwd: s.cwd, label: s.label, status: s.status };
  }
}

module.exports = { SessionRegistry, IDLE_AFTER_MS };
```

```js
// server/pty.js
const pty = require('node-pty');

function spawnClaude(cwd, { command = 'claude', args = [] } = {}) {
  const proc = pty.spawn(command, args, {
    name: 'xterm-color',
    cols: 120,
    rows: 30,
    cwd,
    env: process.env,
  });
  return {
    onData: (cb) => proc.onData(cb),
    onExit: (cb) => proc.onExit(() => cb()),
    write: (data) => proc.write(data),
    resize: (cols, rows) => proc.resize(cols, rows),
    kill: () => proc.kill(),
  };
}

module.exports = { spawnClaude };
```

## Task 5: App server (HTTP static + WebSocket hub)

**Files:**
- Create: `server/app.js`
- Test: `test/app.test.js`

`createApp({ spawnPty })` returns an unstarted `http.Server` with a WebSocket hub wired to a `SessionRegistry`. Injecting `spawnPty` keeps it testable with a fake PTY.

WebSocket protocol:
- client → server: `{type:'create', cwd}`, `{type:'input', id, data}`, `{type:'attach', id}`
- server → client: `{type:'sessions', sessions:[{id,cwd,label,status}]}`, `{type:'output', id, data}`, `{type:'attached', id, buffer}`, `{type:'error', message}`

- [ ] **Step 1: Write the failing test**

```js

function fakePtyFactory() {
  const ptys = [];
  const factory = () => {
    const o = { written: [] };
    o.onData = (cb) => { o._data = cb; };
    o.onExit = (cb) => { o._exit = cb; };
    o.write = (d) => o.written.push(d);
    o.kill = () => {};
    ptys.push(o);
    return o;
  };
  return { factory, ptys };
}

function nextMessage(ws, predicate) {
  return new Promise((resolve) => {
    ws.on('message', function handler(raw) {
      const msg = JSON.parse(raw);
      if (predicate(msg)) { ws.off('message', handler); resolve(msg); }
    });
  });
}

test('create over WS produces a session and streams its output', async () => {
  const { factory, ptys } = fakePtyFactory();
  const { server } = createApp({ spawnPty: factory });
  await new Promise((r) => server.listen(0, '127.0.0.1', r));
  const port = server.address().port;
  const ws = new WebSocket(`ws://127.0.0.1:${port}`);
  await new Promise((r) => ws.on('open', r));

  const sessionsMsg = nextMessage(ws, (m) => m.type === 'sessions' && m.sessions.length === 1);
  ws.send(JSON.stringify({ type: 'create', cwd: 'C:/proj/demo' }));
  const sm = await sessionsMsg;
  assert.strictEqual(sm.sessions[0].label, 'demo');

  const outMsg = nextMessage(ws, (m) => m.type === 'output');
  ptys[0]._data('STREAMED');
  const om = await outMsg;
  assert.strictEqual(om.data, 'STREAMED');

  ws.close();
  await new Promise((r) => server.close(r));
```

```js
// server/app.js
const http = require('node:http');
const fs = require('node:fs');
const path = require('node:path');
const { WebSocketServer } = require('ws');
const { SessionRegistry } = require('./sessions');

const DEFAULT_PUBLIC_DIR = path.join(__dirname, '..', 'public');
const MIME = {
  '.html': 'text/html; charset=utf-8',
  '.js': 'text/javascript; charset=utf-8',
  '.css': 'text/css; charset=utf-8',
};

function createApp({ spawnPty, publicDir = DEFAULT_PUBLIC_DIR } = {}) {
  const registry = new SessionRegistry({ spawnPty });

  const server = http.createServer((req, res) => {
    let urlPath = decodeURIComponent(req.url.split('?')[0]);
    if (urlPath === '/') urlPath = '/index.html';
    const filePath = path.join(publicDir, path.normalize(urlPath));
    if (!filePath.startsWith(publicDir)) {
      res.writeHead(403);
      res.end('forbidden');
      return;
    }
    fs.readFile(filePath, (err, buf) => {
      if (err) {
        res.writeHead(404);
        res.end('not found');
        return;
      }
      res.writeHead(200, { 'content-type': MIME[path.extname(filePath)] || 'application/octet-stream' });
      res.end(buf);
    });
  });

  const wss = new WebSocketServer({ server });
  const clients = new Set();
  const broadcast = (msg) => {
    const text = JSON.stringify(msg);
    for (const c of clients) if (c.readyState === 1) c.send(text);
  };

  registry.on('output', (id, data) => broadcast({ type: 'output', id, data }));
  registry.on('sessions', () => broadcast({ type: 'sessions', sessions: registry.list() }));

  wss.on('connection', (ws) => {
    clients.add(ws);
    ws.send(JSON.stringify({ type: 'sessions', sessions: registry.list() }));
    ws.on('message', (raw) => {
      let m;
      try { m = JSON.parse(raw); } catch { return; }
      if (m.type === 'create') {
        try { registry.create(m.cwd); }
        catch (e) { ws.send(JSON.stringify({ type: 'error', message: String(e && e.message || e) })); }
      } else if (m.type === 'input') {
        registry.write(m.id, m.data);
      } else if (m.type === 'attach') {
        ws.send(JSON.stringify({ type: 'attached', id: m.id, buffer: registry.bufferOf(m.id) }));
      }
    });
    ws.on('close', () => clients.delete(ws));
  });

  const interval = setInterval(() => registry.tickStatus(), 1000);
  interval.unref?.();
  server.on('close', () => clearInterval(interval));

  return { server, registry, wss };
}

module.exports = { createApp };
```

```js
window.addEventListener('resize', () => fit.fit());

let sessions = [];
let focusedId = null;

const listEl = document.getElementById('session-list');
const errorEl = document.getElementById('error');

function render() {
  listEl.innerHTML = '';
  for (const s of sessions) {
    const li = document.createElement('li');
    if (s.id === focusedId) li.className = 'active';
    const dot = document.createElement('span');
    dot.className = `dot ${s.status}`;
    const label = document.createElement('span');
    label.textContent = s.label;
    li.append(dot, label);
    li.onclick = () => focus(s.id);
    listEl.appendChild(li);
  }
}

function focus(id) {
  focusedId = id;
  errorEl.textContent = '';
  ws.send(JSON.stringify({ type: 'attach', id }));
  render();
}

ws.addEventListener('message', (ev) => {
  const m = JSON.parse(ev.data);
  if (m.type === 'sessions') {
    sessions = m.sessions;
    if (!focusedId && sessions.length) focus(sessions[0].id);
    else render();
  } else if (m.type === 'attached' && m.id === focusedId) {
    term.reset();
    term.write(m.buffer);
  } else if (m.type === 'output' && m.id === focusedId) {
    term.write(m.data);
  } else if (m.type === 'error') {
    errorEl.textContent = m.message;
  }
});

term.onData((data) => {
  if (focusedId) ws.send(JSON.stringify({ type: 'input', id: focusedId, data }));
});

document.getElementById('add-btn').onclick = () => {
  const cwd = prompt('Folder to start a Claude Code session in:');
  if (cwd) ws.send(JSON.stringify({ type: 'create', cwd }));
};
```
